server{
        listen 89;
        server_name localhost;
        
        # 对源站点验证
        valid_referers *.imooc.com;
        # 非法引入会进行下方判断
        if($invalid_referer){
        	return 404;
        }

        location / {
            root html;
            index zzm.html index.htm;
        }
    }
