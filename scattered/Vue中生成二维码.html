# Vue中生成二维码

# 1.安装qrcodejs

npm install qrcodejs2

# 2.指定页面引入

import QRCode from 'qrcodejs2'

# 3.定义div模板

<div class="qrcode" ref="qrCodeUrl"></div>

# 4.定义方法

creatQrCode(codeUrl) {
  var qrcode = new QRCode(this.$refs.qrCodeUrl, {
      text: codeUrl,
      width: 100,
      height: 100
 })
}
